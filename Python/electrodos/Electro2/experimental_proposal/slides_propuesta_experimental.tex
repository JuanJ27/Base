\begin{frame}{Fundamento: parámetros \(S\), \(R\), \(T\) y \(A\)}
  \Large
  \[
    S_{ij}(f)=|S_{ij}(f)|e^{i\phi_{ij}(f)}
  \]
  \[
    R=|S_{11}|^2,
    \qquad
    T=|S_{21}|^2,
    \qquad
    A=1-R-T.
  \]

  \vfill
  \normalsize
  \(A=1-R-T\) representa absorción solo si reflexión, transmisión y scattering lateral quedan controlados o incorporados a incertidumbre.
\end{frame}

\begin{frame}{Impedancia efectiva y absorción robusta}
  \Large
  \[
    \Gamma=\frac{Z_{\mathrm{eff}}-Z_0}{Z_{\mathrm{eff}}+Z_0},
    \qquad
    Z_{\mathrm{eff}}\approx Z_0.
  \]

  \vfill
  Con plano metálico posterior:
  \[
    S_{21}\approx0,
    \qquad
    A\simeq1-|S_{11}|^2.
  \]

  \vfill
  \begin{center}
    Adaptación de impedancia + pérdidas internas = absorción resonante esperada.
  \end{center}
\end{frame}

\begin{frame}{Resonancia LC y modelo dispersivo}
  \large
  Celdas sublongitud de onda: \(p\ll\lambda\).
  \[
    f_0=\frac{1}{2\pi\sqrt{LC}},
    \qquad
    Q=\frac{f_0}{\Delta f}.
  \]

  \vfill
  \scriptsize
  \[
    \varepsilon_{\mathrm{eff}}(\omega)=\varepsilon_\infty+
    \frac{F\omega_0^2}{\omega_0^2-\omega^2-i\gamma\omega}.
  \]

  \normalsize
  Dispersión, fase y pérdidas cambiarían cerca de \(\omega_0\); la reconfiguración desplazaría \(f_0\) o modificaría \(Q\).
\end{frame}
