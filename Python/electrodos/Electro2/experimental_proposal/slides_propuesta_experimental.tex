\begin{frame}{8. Fundamento II: reflexión, transmisión y absorción}
  \centering
  {\Huge\(R=|S_{11}|^2\)}\\[0.25cm]
  {\Huge\(T=|S_{21}|^2\)}\\[0.25cm]
  {\Huge\(A=1-R-T\)}

  \vfill
  \begin{block}{Advertencia}
    \centering
    Solo representa absorción si el scattering lateral y la potencia relevante están controlados.
  \end{block}
  \rubrica{3}{Balance de potencia}
\end{frame}

\begin{frame}{9. Fundamento III: impedancia efectiva}
  \centering
  {\Large Impedancia del vacío}\\[0.1cm]
  {\Huge\(Z_0=\sqrt{\mu_0/\varepsilon_0}\approx\SI{377}{\ohm}\)}

  \vfill
  {\Huge\(\Gamma=\dfrac{Z_{\mathrm{eff}}-Z_0}{Z_{\mathrm{eff}}+Z_0}\)}

  \vfill
  {\Large Absorción esperada: \(Z_{\mathrm{eff}}\approx Z_0\) + pérdidas internas}
  \rubrica{3}{Condición física de absorción}
\end{frame}

\begin{frame}{10. Fundamento IV: modelo LC y resonancia}
  \centering
  {\Huge\(p\ll\lambda\)}\\[0.35cm]
  {\Huge\(f_0=\dfrac{1}{2\pi\sqrt{LC}}\)}\\[0.35cm]
  {\Huge\(Q=\dfrac{f_0}{\Delta f}\)}

  \vfill
  {\Large Varactores o diodos modificarían \(C\), pérdidas o estado conductor.}
  \rubrica{3}{Resonancia sublongitud de onda}
\end{frame}

\begin{frame}{11. Fundamento V: Lorentz y supuestos}
  \centering
  {\Large\(
  \varepsilon_{\mathrm{eff}}(\omega)=\varepsilon_\infty+
  \dfrac{F\omega_0^2}{\omega_0^2-\omega^2-i\gamma\omega}
  \)}

  \vfill
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \centering\Large \(d_{\mathrm{FF}}\ge 2D^2/\lambda\)
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering\Large Régimen lineal y calibración válida
    \end{column}
  \end{columns}
  \rubrica{3}{30\% de la evaluación}
\end{frame}

\begin{frame}{20. Discusión, conclusiones y bibliografía mínima}
  \Large
  \begin{itemize}
    \item Pico de \(A\): adaptación \(Z_{\mathrm{eff}}\approx Z_0\) + pérdidas.
    \item \(A<1\): reflexión, transmisión, bordes o pérdidas insuficientes.
    \item \(f_0\) desplazado: tolerancias, sustrato o reconfiguración.
  \end{itemize}
