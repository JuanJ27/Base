\begin{frame}
  \titlepage
  \vspace{-0.5cm}
  \begin{block}{Pregunta experimental}
    ¿Cómo determinar $e/m$ midiendo $V$, $B$ y el radio $r$ de un haz electrónico curvado?
  \end{block}
\end{frame}

\begin{frame}{Motivación}
  \Large
  Un campo magnético transversal permite convertir la trayectoria de un haz cargado en una medición cuantitativa.

  \vfill
  \begin{columns}[T]

      \end{tikzpicture}

      \begin{itemize}
        \item Observable directo: curvatura del haz.

        \item Resultado: propiedad fundamental del electrón.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Objetivos}
  \Large
  \begin{enumerate}
    \item Determinar experimentalmente $e/m$.
    \item Calibrar $B(I)$ para bobinas de Helmholtz.
    \item Medir $r$ para varios pares $(V,I)$.
    \item Comparar con
      \[
        \left(\frac{e}{m}\right)_\text{aceptado}
        =\SI{1.758820e11}{\coulomb\per\kilogram}.
      \]
  \end{enumerate}
\end{frame}

\begin{frame}{Fundamento físico}
  \Large
  Aceleración eléctrica:
  \[
    eV=\frac{1}{2}mv^2
  \]

  \vfill
  Fuerza magnética:
  \[
    \vec F=q\,\vec v\times\vec B,
    \qquad |\vec v\perp\vec B|:\quad F=e v B
  \]

  \vfill
  Movimiento circular:
  \[
    e v B=\frac{mv^2}{r}
  \]
\end{frame}

\begin{frame}{Resultado clave}
  \centering
  \Huge
  \[
    \boxed{\frac{e}{m}=\frac{2V}{B^2r^2}}
  \]

  \vfill
  \Large
  Tres mediciones controlan el experimento:
  \[
    V \quad B(I) \quad r
  \]

  \vfill
  \normalsize
  El signo de la carga no entra en la magnitud $e/m$; la curvatura fija la dinámica.
\end{frame}

\begin{frame}{Procedimiento resumido}
  \Large
  \begin{enumerate}
    \item Verificar conexiones, seguridad y rango de operación.
    \item Estabilizar emisión del haz.
    \item Fijar $V$ y variar $I$.
    \item Medir $r$ para cada trayectoria circular estable.
    \item Repetir para varios voltajes.
    \item Calcular $B(I)$, $e/m$ e incertidumbres.
  \end{enumerate}
\end{frame}

\begin{frame}{Incertidumbres, errores y seguridad}
  \Large
  \[
    \frac{\sigma_{e/m}}{e/m}=\sqrt{
    \left(\frac{\sigma_V}{V}\right)^2+
    \left(2\frac{\sigma_B}{B}\right)^2+
    \left(2\frac{\sigma_r}{r}\right)^2}
  \]

  \vfill
  \begin{itemize}
    \item Dominantes: lectura de $r$, calibración de $B$, alineación.
    \item Sistemáticos: campo terrestre, paralaje, no uniformidad.
    \item Seguridad: alta tensión, vidrio, calentamiento de bobinas.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusión}
  \Large
  Se mide:
  \[
    \frac{e}{m}\quad \text{a partir de}\quad V,\ B(I),\ r
  \]

  \vfill
  Se valida con:
  \begin{itemize}
    \item consistencia lineal de $B$ con $I$;
    \item ajuste de $V$ contra $B^2r^2$;
    \item comparación con el valor aceptado dentro de incertidumbres.
  \end{itemize}

  \vfill
  \centering
  \normalsize
  Alcance: deflexión magnética de electrones en tubo e/m; no medición de radiación sincrotrón real.
\end{frame}
